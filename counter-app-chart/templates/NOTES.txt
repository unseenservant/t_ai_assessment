Thank you for installing {{ .Chart.Name }}.

Your release is named {{ .Release.Name }}.

To access the Counter App:

{{- if contains "NodePort" .Values.app.service.type }}
  1. Get the application URL by running these commands:
     export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Values.app.name }})
     export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
     echo http://$NODE_IP:$NODE_PORT

{{- else if contains "LoadBalancer" .Values.app.service.type }}
  1. Get the application URL by running these commands:
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ .Values.app.name }}'
     export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ .Values.app.name }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
     echo http://$SERVICE_IP:{{ .Values.app.service.port }}

{{- else if contains "ClusterIP" .Values.app.service.type }}
  1. Get the application URL by running these commands:
     export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app={{ .Values.app.name }}" -o jsonpath="{.items[0].metadata.name}")
     kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:{{ .Values.app.service.targetPort }}
     Visit http://127.0.0.1:8080 to use your application

{{- else if contains "NodePort" .Values.app.service.type }}
  1. If you're using minikube, you can access the application with:
     minikube service {{ .Values.app.name }}
{{- end }}

Health Check Endpoint:
To check the health of the application, you can access the health endpoint:
{{- if contains "NodePort" .Values.app.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Values.app.name }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  curl http://$NODE_IP:$NODE_PORT/health
{{- else if contains "LoadBalancer" .Values.app.service.type }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ .Values.app.name }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  curl http://$SERVICE_IP:{{ .Values.app.service.port }}/health
{{- else if contains "ClusterIP" .Values.app.service.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:{{ .Values.app.service.targetPort }}
  curl http://127.0.0.1:8080/health
{{- end }}
